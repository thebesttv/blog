#+BEGIN_SRC elisp :exports both
  (let ((available-files
         (directory-files-recursively "." (rx ".org" eos)))
        res)
    (dolist (file available-files)
      (setq res (concat res (format "%s\n" file))))
    (print res))
#+END_src
